    (function() {
        'use strict';

        // Selection System
        let selectedLeads = new Set();
        
        window.toggleSelection = function(checkbox) {
            const leadCard = checkbox.closest('[data-lead-id]');
            const leadId = leadCard.dataset.leadId;
            
            if (checkbox.checked) {
                selectedLeads.add(leadId);
                leadCard.classList.add('selection-overlay');
            } else {
                selectedLeads.delete(leadId);
                leadCard.classList.remove('selection-overlay');
            }
            
            updateBulkActionBar();
        };
        
        function updateBulkActionBar() {
            const bulkBar = document.getElementById('bulkActionBar');
            const countEl = document.getElementById('selectedCount');
            
            if (selectedLeads.size &gt; 0) {
                bulkBar.classList.remove('hidden');
                countEl.textContent = `${selectedLeads.size} leads selecionados`;
            } else {
                bulkBar.classList.add('hidden');
            }
        }
        
        window.clearSelection = function() {
            document.querySelectorAll('.lead-checkbox').forEach(checkbox =&gt; {
                checkbox.checked = false;
                checkbox.closest('[data-lead-id]').classList.remove('selection-overlay');
            });
            selectedLeads.clear();
            updateBulkActionBar();
        };

        // Mobile Chat Toggle
        window.toggleMobileChat = function() {
            const modal = document.getElementById('mobileChatModal');
            modal.classList.toggle('translate-x-full');
        };

        // Notification Toggle
        let notificationDropdown = null;
        
        document.addEventListener('click', function(e) {
            if (e.target.closest('.fa-bell')) {
                e.preventDefault();
                const dropdown = document.getElementById('notificationDropdown');
                dropdown.classList.toggle('hidden');
                notificationDropdown = dropdown.classList.contains('hidden') ? null : dropdown;
            } else if (notificationDropdown &amp;&amp; !e.target.closest('#notificationDropdown')) {
                notificationDropdown.classList.add('hidden');
                notificationDropdown = null;
            }
        });

        // Lead Card Interactions
        document.addEventListener('click', function(e) {
            const leadCard = e.target.closest('[data-lead-id]');
            if (!leadCard) return;
            
            // Prevent selection when clicking action buttons
            if (e.target.closest('button') || e.target.closest('.lead-checkbox')) return;
            
            // Simulate lead detail view
            console.log('Opening lead details for:', leadCard.dataset.leadId);
            // Here you would open a modal or navigate to detail page
        });

        // Infinite Scroll Simulation
        let isLoading = false;
        window.addEventListener('scroll', function() {
            if (isLoading) return;
            
            const scrollPosition = window.innerHeight + window.scrollY;
            const documentHeight = document.documentElement.offsetHeight;
            
            if (scrollPosition &gt;= documentHeight - 1000) {
                isLoading = true;
                // Simulate loading more leads
                setTimeout(() =&gt; {
                    console.log('Loading more leads...');
                    isLoading = false;
                }, 1000);
            }
        });

        // Simulate real-time updates
        setInterval(() =&gt; {
            // Update notification badges randomly
            const badges = document.querySelectorAll('.bg-red-500');
            badges.forEach(badge =&gt; {
                const currentCount = parseInt(badge.textContent) || 0;
                if (Math.random() &gt; 0.9) { // 10% chance to update
                    badge.textContent = currentCount + 1;
                }
            });
        }, 10000); // Every 10 seconds

        // Lead action handlers
        document.addEventListener('click', function(e) {
            const button = e.target.closest('[data-event]');
            if (!button) return;
            
            const event = button.dataset.event;
            
            switch(event) {
                case 'click:connectLead':
                    console.log('Connecting to lead...');
                    // Show success notification
                    showNotification('ConexÃ£o enviada com sucesso!', 'success');
                    break;
                    
                case 'click:openWhatsApp':
                    console.log('Opening WhatsApp...');
                    // Simulate WhatsApp opening
                    showNotification('Abrindo WhatsApp...', 'info');
                    break;
                    
                case 'click:sendEmail':
                    console.log('Opening email composer...');
                    showNotification('Abrindo composer de email...', 'info');
                    break;
                    
                case 'click:saveLead':
                    console.log('Saving lead to favorites...');
                    showNotification('Lead salvo nos favoritos!', 'success');
                    break;
                    
                case 'click:bulkWhatsApp':
                    console.log('Bulk WhatsApp action...');
                    showNotification(`Enviando WhatsApp para ${selectedLeads.size} leads...`, 'info');
                    break;
                    
                case 'click:bulkEmail':
                    console.log('Bulk email action...');
                    showNotification(`Enviando email para ${selectedLeads.size} leads...`, 'info');
                    break;
                    
                case 'click:loadMoreLeads':
                    console.log('Loading more leads...');
                    button.innerHTML = '&lt;i class=&quot;fas fa-spinner fa-spin mr-2&quot;&gt;&lt;/i&gt;Carregando...';

                        button.innerHTML = '&lt;i class=&quot;fas fa-plus mr-2&quot;&gt;&lt;/i&gt;Carregar mais leads';

            let selectedLeads = new Set();
            let selectionMode = false;

            // Toggle lead selection
            window.toggleLeadSelection = (leadId) =&gt; {
                if (selectedLeads.has(leadId)) {
                    selectedLeads.delete(leadId);
                } else {
                    selectedLeads.add(leadId);
                }
                updateSelectionUI();
                
                // Add visual feedback
                const card = document.querySelector(`[data-lead-id=&quot;${leadId}&quot;]`);
                if (card) {
                    if (selectedLeads.has(leadId)) {
                        card.classList.add('ring-2', 'ring-facebook-blue', 'selection-animation');
                    } else {
                        card.classList.remove('ring-2', 'ring-facebook-blue');
                    }
                }
            };

            // Toggle selection mode
            window.toggleSelectionMode = () =&gt; {
                selectionMode = !selectionMode;
                const checkboxes = document.querySelectorAll('.lead-checkbox');
                const cards = document.querySelectorAll('[data-lead-id]');
                
                checkboxes.forEach(checkbox =&gt; {
                    checkbox.classList.toggle('hidden', !selectionMode);
                });

                if (selectionMode) {
                    cards.forEach(card =&gt; {
                        card.classList.add('cursor-pointer');
                        card.addEventListener('click', handleCardClick);
                    });
                } else {
                    clearSelection();
                    cards.forEach(card =&gt; {
                        card.removeEventListener('click', handleCardClick);
                    });
                }
            };

            // Handle card click in selection mode
            function handleCardClick(event) {
                if (!selectionMode) return;
                
                const leadId = parseInt(event.currentTarget.getAttribute('data-lead-id'));
                const checkbox = event.currentTarget.querySelector('.lead-checkbox');
                
                checkbox.checked = !checkbox.checked;
                toggleLeadSelection(leadId);
            }

            // Update selection UI
            function updateSelectionUI() {
                const count = selectedLeads.size;
                const selectionBar = document.getElementById('selectionBar');
                const selectedCount = document.getElementById('selectedCount');
                
                selectedCount.textContent = `${count} lead${count !== 1 ? 's' : ''} selecionado${count !== 1 ? 's' : ''}`;
                
                if (count &gt; 0) {
                    selectionBar.classList.remove('translate-y-20');
                } else {
                    selectionBar.classList.add('translate-y-20');
                }
            }

            // Clear selection
            window.clearSelection = () =&gt; {
                selectedLeads.clear();
                const checkboxes = document.querySelectorAll('.lead-checkbox');
                const cards = document.querySelectorAll('[data-lead-id]');
                
                checkboxes.forEach(checkbox =&gt; {
                    checkbox.checked = false;
                });
                
                cards.forEach(card =&gt; {
                    card.classList.remove('ring-2', 'ring-facebook-blue');
                });
                
                updateSelectionUI();
                selectionMode = false;
                
                checkboxes.forEach(checkbox =&gt; {
                    checkbox.classList.add('hidden');
                });
            };

            // Toggle mobile chat
            window.toggleChat = () =&gt; {
                const mobileChat = document.getElementById('mobileChat');
                mobileChat.classList.toggle('translate-x-full');
            };

            // Infinite scroll simulation
            let loading = false;
            function loadMoreLeads() {
                if (loading) return;
                loading = true;
                
                setTimeout(() =&gt; {
                    // Simulate loading more leads
                    console.log('Loading more leads...');
                    loading = false;
                }, 1000);
            }

            // Scroll event listener
            window.addEventListener('scroll', () =&gt; {
                if ((window.innerHeight + window.scrollY) &gt;= document.body.offsetHeight - 1000) {
                    loadMoreLeads();
                }
            });

            // Initialize tooltips and interactions
            document.addEventListener('DOMContentLoaded', () =&gt; {
                // Add click handlers to action buttons
                const actionButtons = document.querySelectorAll('article button');
                actionButtons.forEach(button =&gt; {
                    button.addEventListener('click', (e) =&gt; {
                        e.stopPropagation();
                        const action = button.textContent.trim();
                        console.log(`Action: ${action}`);
                        
                        // Add visual feedback
                        button.classList.add('scale-95');
                        setTimeout(() =&gt; {
                            button.classList.remove('scale-95');
                        }, 150);
                    });

                setInterval(() =&gt; {

            // Multi-select functionality
            let selectedLeads = new Set();
            const floatingActionBar = document.getElementById('floatingActionBar');
            const selectedCount = document.getElementById('selectedCount');
            const clearSelection = document.getElementById('clearSelection');

            // Handle lead selection
            document.addEventListener('click', function(e) {
                if (e.target.closest('.select-lead-btn')) {
                    const button = e.target.closest('.select-lead-btn');
                    const leadId = button.dataset.leadId;
                    const checkbox = document.querySelector(`input[data-lead-id=&quot;${leadId}&quot;]`);
                    const icon = button.querySelector('i');
                    
                    if (selectedLeads.has(leadId)) {
                        selectedLeads.delete(leadId);
                        icon.className = 'far fa-square text-gray-400';
                        checkbox.checked = false;
                    } else {
                        selectedLeads.add(leadId);
                        icon.className = 'fas fa-check-square text-facebook';
                        checkbox.checked = true;
                    }
                    
                    updateFloatingActionBar();
                }
            });

            function updateFloatingActionBar() {
                const count = selectedLeads.size;
                if (count &gt; 0) {
                    selectedCount.textContent = `${count} selecionado${count &gt; 1 ? 's' : ''}`;
                    floatingActionBar.classList.remove('hidden');
                } else {
                    floatingActionBar.classList.add('hidden');
                }
            }

            // Clear selection
            clearSelection.addEventListener('click', function() {
                selectedLeads.clear();
                document.querySelectorAll('.select-lead-btn i').forEach(icon =&gt; {
                    icon.className = 'far fa-square text-gray-400';
                });
                document.querySelectorAll('.lead-checkbox').forEach(checkbox =&gt; {
                    checkbox.checked = false;
                });
                updateFloatingActionBar();
            });
